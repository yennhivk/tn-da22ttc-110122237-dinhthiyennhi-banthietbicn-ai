📝 Ví dụ sử dụng Permission Checker

📢 Lưu ý:

File này là ví dụ minh họa. Để sử dụng trong trang thực tế, include file permissions-checker.js vào HTML của bạn.

1️⃣ Ẩn/hiện button theo quyền 

// Ẩn/hiện button tạo đơn hàng
permissionChecker.toggleButton('btn-create-order', 'create_orders');

// Ẩn/hiện button xóa đơn hàng
permissionChecker.toggleButton('btn-delete-order', 'delete_orders');

2️⃣ Kiểm tra quyền trong JavaScript 

function testPermission() {
    if (hasPermission('create_orders')) {
        alert('✅ Bạn có quyền tạo đơn hàng');
        // Thực hiện logic tạo đơn
    } else {
        alert('❌ Bạn không có quyền tạo đơn hàng');
    }
}

3️⃣ Menu điều kiện theo quyền

📦 Quản lý đơn hàng
📱 Quản lý sản phẩm
📊 Báo cáo tài chính
⚙️ Cài đặt hệ thống 
// Ẩn/hiện menu items
permissionChecker.toggleElement('menu-orders', 'view_orders');
permissionChecker.toggleElement('menu-products', 'view_products');
permissionChecker.toggleElement('menu-reports', 'view_reports');
permissionChecker.toggleElement('menu-settings', 'edit_settings');

4️⃣ Hiển thị badge quyền 

// Tạo badges cho các quyền
const badges = document.getElementById('permission-badges');
badges.innerHTML = 
    permissionChecker.createBadge('view_orders', 'Xem đơn') +
    permissionChecker.createBadge('create_orders', 'Tạo đơn') +
    permissionChecker.createBadge('edit_orders', 'Sửa đơn') +
    permissionChecker.createBadge('delete_orders', 'Xóa đơn');

5️⃣ Các phương thức có sẵn 

// Load quyền từ server
await permissionChecker.load();

// Kiểm tra 1 quyền
if (permissionChecker.has('create_orders')) { ... }

// Kiểm tra có ít nhất 1 quyền
if (permissionChecker.hasAny(['view_orders', 'create_orders'])) { ... }

// Kiểm tra có tất cả quyền
if (permissionChecker.hasAll(['view_orders', 'edit_orders'])) { ... }

// Ẩn/hiện element
permissionChecker.toggleElement('elementId', 'permission_name');

// Ẩn/hiện nhiều elements
permissionChecker.toggleElements('.btn-delete', 'delete_permission');

// Enable/disable button
permissionChecker.toggleButton('btnId', 'permission_name');

// Debug - xem tất cả quyền
permissionChecker.debug();

// Export quyền ra CSV
permissionChecker.exportToCSV();

// Clear cache
permissionChecker.clear();

// Helper functions (shorthand)
if (hasPermission('create_orders')) { ... }
if (hasAnyPermission(['perm1', 'perm2'])) { ... }
if (hasAllPermissions(['perm1', 'perm2'])) { ... }

6️⃣ Tích hợp vào admin.html 

<!-- Thêm vào head -->
<script src="../js/permissions-checker.js"></script>

<script>
// Sau khi load xong permissions
window.addEventListener('load', async () => {
    await permissionChecker.load();
    
    // Ẩn/hiện các chức năng theo quyền
    setupPermissions();
});

function setupPermissions() {
    // Ẩn/hiện buttons
    permissionChecker.toggleButton('btn-add-product', 'add_product');
    permissionChecker.toggleButton('btn-delete-product', 'delete_product');
    
    // Ẩn/hiện menu items
    permissionChecker.toggleElement('menu-settings', 'edit_settings');
    permissionChecker.toggleElement('menu-reports', 'view_reports');
    
    // Ẩn/hiện tất cả nút xóa
    permissionChecker.toggleElements('.btn-delete', 'delete_permission');
}

// Kiểm tra trước khi thực hiện action
async function deleteProduct(id) {
    if (!hasPermission('delete_product')) {
        alert('Bạn không có quyền xóa sản phẩm');
        return;
    }
    
    // Logic xóa sản phẩm
    await apiCall('/admin/products/' + id, 'DELETE');
}
</script>
